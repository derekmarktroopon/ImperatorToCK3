
language_gaulish = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_gaulish
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_gaulish }
			multiply = 10
		}
	}
	
	color = rgb{ 4 176 101 }
}
