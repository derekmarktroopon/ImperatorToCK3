


look_for_egyptian_tomb_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_activity.dds"
	}
	
	sort_order = 75

	desc = look_for_egyptian_tomb_decision_desc
	selection_tooltip = look_for_egyptian_tomb_decision_tooltip

	cooldown = { days = 365 }

    is_shown = {
    	is_ai = no
    	NOR = {
    		has_variable =  searched_for_the_pharaohs_tombs 
    		has_global_variable = ancient_egyptian_tomb_found 
    		culture = culture:predynastic_egyptian
    		culture = culture:middle_egyptian
    		culture = culture:ancient_egyptian
    	}
    	AND = {
    	    OR = {
    	        any_county_in_region = {
                    region = world_africa_north_east
    	        	this = root.capital_county
    	        	count >= 1
    	        }	
    	    	culture = culture:coptic_culture
    	    	culture = culture:egyptian
    	    }    
    	    OR = {
    	    	knows_language = language_coptic
    	    	knows_language = language_kemetic
    	    	learning > 10
    	    }
    	}
    }    

	effect = {
		ROOT = {
			set_variable = {
				name = searching_for_the_pharaohs_tombs
				value = 1
			}
		}
		trigger_event = {
			id = kemetic_revival_event.1
	    }
	}
	
	ai_check_interval = 60

	ai_potential = {
		always = no
	}
}

study_the_heiroglyphs_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/tgp_poet.dds"
	}
	
	sort_order = 75

	desc = study_the_heiroglyphs_decision_desc
	selection_tooltip = study_the_heiroglyphs_decision_tooltip

    is_shown = {
    	is_ai = no
    	has_global_variable = ancient_egyptian_tomb_found 
    	NOT = {
    		OR = {
    			knows_language = language_egyptian
    			has_variable = failed_to_dicipher_heiroglyphs
    			has_variable = studying_heiroglyphs		
    		}
    	}
    	any_directly_owned_province = {
    		any_county_in_region = {
                region = ghw_region_egypt
    		}	    
    	}
    }

	effect = {
		ROOT = {
			set_variable = {
				name = studying_heiroglyphs
				value = 1
			}
		}
		trigger_event = {
			id = kemetic_revival_event.9
			days = { 1 2 }
		}
	}

	ai_check_interval = 60

	ai_potential = {
		always = no
	}
}
