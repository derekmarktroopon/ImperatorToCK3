
unite_anglosaxon_cultures_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_destiny_goal.dds"
	}
    decision_group_type = major
	
    ai_check_interval = 36
	
    desc = unite_anglosaxon_cultures_decision_desc
    confirm_text = unite_anglosaxon_cultures_confirm
	
    is_shown = {
		is_landed = yes
        root_culture = {
        	NOR = {
        		root.culture = culture:anglo_saxon
        		any_parent_culture_or_above = {
					this = culture:anglo_saxon
				}
        	}
        	has_cultural_pillar = heritage_west_germanic
        	OR = {
        		has_name_list = name_list_northumbrian
        		has_name_list = name_list_mercian
        		has_name_list = name_list_east_anglian
        		has_name_list = name_list_west_saxon
        		has_name_list = name_list_east_saxon
        		has_name_list = name_list_jutish
        	}
        }
        current_date >= 820.1.1
		NOT = { has_global_variable = anglo_saxon_cultural_unification }
		NOT = {
			culture:anglo_saxon = {
				any_culture_county = {
					count >= 1
					any_de_jure_top_liege = { this = title:k_england }
				}
			}
		}
    }
	
    is_valid = {
		OR = {
			AND = {
				highest_held_title_tier >= tier_kingdom
				has_title = title:k_england
				save_temporary_scope_as = effect_root
			}
		}
		prestige_level >= high_prestige_level
		realm_size >= 35
    }
	
	is_valid_showing_failures_only = {
		is_imprisoned = no
		is_available_adult = yes
	}
	
    effect = {
    	set_global_variable = {
			name = anglo_saxon_cultural_unification
			value = yes
		}
		custom_tooltip = unite_anglosaxon_cultures.preview.tt
		give_nickname = nick_the_great
		trigger_event = {
			id = anglosaxon_culture_unification.1
		}
    }

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}

}
