
norse_emergence_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_destiny_goal.dds"
	}
    decision_group_type = major
	
    ai_check_interval = 36
	
    desc = norse_emergence_decision_desc
    confirm_text = norse_emergence_start_confirm
	
    is_shown = {
		is_ai = yes
		has_global_variable = vendel_period_began
		NOT = { has_global_variable = norse_emergence_occured }
		current_year >= 780
    }
	
    is_valid = {
		is_ai = yes
		current_year >= 780
    }
	
	is_valid_showing_failures_only = {
		is_ai = yes
	}
	
    effect = {
		set_global_variable = {
			name = norse_emergence_occured
			value = yes
		}
		trigger_event = {
			id = proto_germanic_divergence_events.2
		}
    }

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}

}
