
venetic_divergence_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_destiny_goal.dds"
	}
    decision_group_type = major
	
    ai_check_interval = 36
	
    desc = venetic_divergence_decision_desc
    confirm_text = venetic_divergence_confirm
	
    is_shown = {
		is_ai = yes
		NOT = { has_global_variable = venetic_divergence }
		current_year >= 500
		current_year < 800
    }
	
    is_valid = {
		is_ai = yes
		current_year >= 500
		current_year < 800
    }
	
	is_valid_showing_failures_only = {
		is_ai = yes
	}
	
    effect = {
		set_global_variable = {
			name = venetic_divergence
			value = yes
		}
		trigger_event = {
			id = venetic_divergence_events.2
		}
    }

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}

}
