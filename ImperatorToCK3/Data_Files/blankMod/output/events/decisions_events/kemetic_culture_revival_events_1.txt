namespace = kemetic_revival_event

kemetic_revival_event.1 = { #whispers of the nile
	title = kemetic_revival_event.1.t
	desc = kemetic_revival_event.1.desc

	left_portrait = {
		character = root
		animation = idle
	}
	theme = hunting
	override_background = {
		reference = wilderness_desert
	}

	trigger = {
	    AND = {
	    is_ai = no
		NOT = {
			OR = {
				has_variable =  searched_for_the_pharaohs_tombs 
				has_global_variable = ancient_egyptian_tomb_found 
			}
		}
		any_held_county = { 
		    AND = {
			    geographical_region = custom_roman_aegyptus
			}
		}

		OR = {
			AND = {
				OR ={
				knows_language = language_coptic
				knows_language = language_egyptian
				}
				learning > 5
				}
				learning > 10
			}
		}
	}
	option = {
		name = kemetic_revival_event.1.a #
		custom_tooltip = kemetic_revival_event.1.adesc #You start looking for ancient structures
		hidden_effect = {
			random_list = {
				25 = {
					trigger_event = {
						id = kemetic_revival_event.3
						days = { 30 100 }
					}
					modifier = {
						add = 20
						learning > 10
					}
					modifier = {
						add = 30
						learning > 15
					}
					modifier = {
						add = 40
						learning > 20
					}
					modifier = {
						add = 50
						learning > 30
					}

				}
				75 = {
					trigger_event = {
						id = kemetic_revival_event.2
						days = { 100 300 }
					}	
				}
			}
		}
	}

	option = {
		name = kemetic_revival_event.1.b #what a waste of time!
		custom_tooltip = kemetic_revival_event.1.b.desc #You carry on with your life as usual
		remove_variable = searching_for_the_pharaohs_tombs
			set_variable = { 
				name = searched_for_the_pharaohs_tombs 
				value = root
			}       #set_global_variable = { name = ancient_egyptian_tomb_found  value = flag:ancient_egyptian_tomb_found } 
	}

}

kemetic_revival_event.7 = { #heiroglyphs diciphered!
	title = kemetic_revival_event.7.t 
	desc = kemetic_revival_event.7.desc
	left_portrait = {
		character = root
		animation = throne_room_cheer_1
	}
	right_portrait = {
		character = scope:court_chaplain
		animation = throne_room_cheer_2
	}
	theme = learning
	override_background = {
		reference = study
	}
	immediate = {
		add_prestige = 350
		remove_variable = studying_heiroglyphs
	}
	option = {
		name = kemetic_revival_event.7.a #learn kemetic
		learn_language_of_culture = culture:middle_egyptian
	}
	option = {
		name = kemetic_revival_event.7.b #adopt kemetic culture
	    create_hybrid_culture = culture:middle_egyptian
	}
	option = {
		name = kemetic_revival_event.7.c #adopt neo kemetic culture
	    hidden_effect = {
	    	root.culture = {
	    		change_cultural_acceptance = {
					target = culture:neo_kemetic
					value = 100
					desc = debug
				}
	    	}
	    	if = {
				limit = { any_spouse = { is_playable_character = no } }
				every_spouse = {
					limit = {
						is_playable_character = no
						culture = root.culture
					}
					set_culture = culture:neo_kemetic
				}
			}
			if = {
				limit = {
					any_close_family_member = {
						is_playable_character = no
						NOT = { is_spouse_of = root }
					}
				}
				every_close_family_member = {
					limit = {
						is_playable_character = no
						NOT = { is_spouse_of = root }
					}
					custom = all_unlanded_family_members
					set_culture = culture:neo_kemetic
				}
			}
			if = {
				limit = {
					any_child = {
						NOT = { is_spouse_of = root }
					}
				}
				every_child = {
					limit = {
						NOT = { is_spouse_of = root }
					}
					custom = all_children_custom
					set_culture = culture:neo_kemetic
				}
			}
			every_courtier = {
				limit = {
					has_culture = root.culture
					dynasty = root.dynasty
				}
				set_culture = culture:neo_kemetic
			}
			capital_county = { set_county_culture = culture:neo_kemetic }
			random_sub_realm_county = {
				limit = {
					title_province = { geographical_region = world_africa_east }
					NOR = { 
						this = capital_county
						has_variable = county_flipped_to_neo_kemetic 
					}
				}
				set_county_culture = culture:neo_kemetic
				set_variable = {
					name = county_flipped_to_neo_kemetic
					value = 0
				}
			}
			random_sub_realm_county = {
				limit = {
					title_province = { geographical_region = world_africa_east }
					NOR = { 
						this = capital_county
						has_variable = county_flipped_to_neo_kemetic 
					}
				}
				set_county_culture = culture:neo_kemetic
				set_variable = {
					name = county_flipped_to_neo_kemetic
					value = 0
				}
			}
			random_sub_realm_county = {
				limit = {
					title_province = { geographical_region = world_africa_east }
					NOR = { 
						this = capital_county
						has_variable = county_flipped_to_neo_kemetic 
					}
				}
				set_county_culture = culture:neo_kemetic
				set_variable = {
					name = county_flipped_to_neo_kemetic
					value = 0
				}
			}
			random_sub_realm_county = {
				limit = {
					title_province = { geographical_region = world_africa_east }
					NOR = { 
						this = capital_county
						has_variable = county_flipped_to_neo_kemetic 
					}
				}
				set_county_culture = culture:neo_kemetic
				set_variable = {
					name = county_flipped_to_neo_kemetic
					value = 0
				}
			}
			culture:neo_kemetic = {
				reset_culture_creation_date = yes
				get_all_innovations_from = culture:middle_egyptian
				get_all_innovations_from = culture:coptic_culture
				get_all_innovations_from = culture:egyptian
				get_all_innovations_from = root.culture
			}
		}
		set_culture = culture:neo_kemetic
		if = {
			limit = {
				NOT = { has_trait = eccentric }
			}
			add_trait = eccentric
		}	
	}
	option = {
		name = kemetic_revival_event.7.d #adopt kemetic culture wholesale
		trigger = {
			OR = {
				has_trait = eccentric
				has_trait = lunatic_1
			}
		}
	    hidden_effect = {
	    	root.culture = {
	    		change_cultural_acceptance = {
					target = culture:middle_egyptian
					value = 20
					desc = debug
				}
	    	}
			every_courtier = {
				limit = {
					has_culture = root.culture
					dynasty = root.dynasty
				}
				set_culture = culture:middle_egyptian
			}
			capital_county = { set_county_culture = culture:middle_egyptian }
		}
		set_culture = culture:middle_egyptian
		if = {
			limit = {
				NOT = { has_trait = lunatic_1 }
			}
			add_trait = lunatic_1
		}	
	}

	option = {
		name = kemetic_revival_event.7.e #adopt kemetic culture and faith
		trigger = {
			AND = {
				has_trait = lunatic_1
				has_trait = possessed
			}
		}
	    hidden_effect = {
	    	root.culture = {
	    		change_cultural_acceptance = {
					target = culture:middle_egyptian
					value = 20
					desc = debug
				}
	    	}
			every_courtier = {
				limit = {
					has_culture = root.culture
					dynasty = root.dynasty
				}
				set_culture = culture:middle_egyptian
				set_character_faith = faith:egyptian_pagan
			}
			capital_county = { set_county_culture = culture:middle_egyptian }
			capital_county = { set_county_faith = faith:egyptian_pagan }
		}
		set_culture = culture:middle_egyptian
		set_character_faith = faith:egyptian_pagan
		scope:court_chaplain = {
			death = {
				death_reason = death_head_ripped_off
				killer = ROOT
			}
		}
		custom_tooltip = kemetic_revival_event.7.edesc
	}

}
