gallic_divergence_events.1 = {
hidden = yes

	immediate = {
		every_culture_global = {
			if = {
				limit = {
					has_name_list = name_list_gaulish
				}
				every_culture_county = { #Northern France
					limit = {
						OR = {
							title_province = { duchy = title:d_brittany }

						}
					}
					add_to_list = gaulish_cult_count_for_france

				if = {
					limit = {
						any_culture_global.any_culture_county = {
							is_in_list = gaulish_cult_count_for_france
							count >= 1
						}
					}
					random_culture_county = {
						limit = {
							is_in_list = gaulish_cult_count_for_france
						}
						holder = {
							save_temporary_scope_as = holder_of_gaulish_county
							set_culture = culture:gaulish
							create_divergent_culture = yes
                            scope:holder_of_gaulish_county.culture = {
								save_scope_as = gaulish_french_culture
								set_variable = {
									name = recent_event_divergence
									#value = true
									days = 3650
								}
								set_name_list = name_list_neo_gaule
							}
						}
					}
					every_in_list = {#####
						list = gaulish_cult_count_for_france

							count >= 1
						}
					}
					random_culture_county = {
						limit = {
							is_in_list = gaulish_cult_count_for_occitania

							save_temporary_scope_as = holder_of_gaulish_county
