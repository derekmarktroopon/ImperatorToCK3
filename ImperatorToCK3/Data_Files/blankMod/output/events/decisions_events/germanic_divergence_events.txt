	hidden = yes

	immediate = {
		every_culture_global = {
			limit = {
				OR = {
					has_name_list = name_list_proto_norse
					has_name_list = name_list_later_vendel
				}
			}
			every_culture_county = {
				limit = {
					geographical_region = world_europe_north
				}
				add_to_list = vendel_counties_to_make_old_norse
			}
		}
		every_in_list = {#####
			list = vendel_counties_to_make_old_norse
			set_county_culture = culture:norse
			holder = {
				if = {
					limit = {
						culture = { 
							OR = {
								has_name_list = name_list_proto_norse
								has_name_list = name_list_later_vendel
							}
						}
						capital_province = {
							geographical_region = world_europe_north	
						}
					}
					set_culture = culture:norse
					every_courtier = {
						limit = {
							culture = { 
								OR = {
									has_name_list = name_list_proto_norse
									has_name_list = name_list_later_vendel
								}
							}
						}
						set_culture = culture:norse
					}
				}
			}
		}
		culture:norse = {
			reset_culture_creation_date = yes
			get_all_innovations_from = culture:proto_norse
		}
	}
}
