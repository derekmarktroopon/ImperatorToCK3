# Romano-Venetic becomes Venetian
##################################################

venetic_divergence_events.1 = { 
	hidden = yes

	immediate = {
		every_culture_global = {
			limit = {
				has_name_list = name_list_romanoveneti
			}
			every_culture_county = {
				limit = {
					geographical_region = world_europe_south
				}
				add_to_list = romanoveneti_counties_to_make_venetian
			}
		}
		every_in_list = {#####
			list = romanoveneti_counties_to_make_venetian
			set_county_culture = culture:venetian
			holder = {
				if = {
					limit = {
						culture = { 
							has_name_list = name_list_romanoveneti
						}
						capital_province = {
							geographical_region = world_europe_south	
						}
					}
					set_culture = culture:venetian
					every_courtier = {
						limit = {
							culture = { 
								has_name_list = name_list_romanoveneti
							}
						}
						set_culture = culture:venetian
					}
				}
			}
		}
		culture:venetian = {
			reset_culture_creation_date = yes
			get_all_innovations_from = culture:romanoveneti
		}
	}
}
